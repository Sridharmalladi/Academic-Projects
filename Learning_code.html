#the dataset
diabetes_data <- read.csv("diabetes.csv")
diabetes_data$Outcome <- as.factor(diabetes_data$Outcome)

#Splitting the data into training and testing sets
set.seed(123)
train_indices <- sample(1:nrow(diabetes_data), 0.7 * nrow(diabetes_data))
train_data <- diabetes_data[train_indices, ]
test_data <- diabetes_data[-train_indices, ]

#Logistic Regression
logistic_model <- glm(Outcome ~ ., data = train_data, family = "binomial")
logistic_predictions <- predict(logistic_model, newdata = test_data, type = "response")

#Convert predictions to binary
logistic_predictions_binary <- factor(ifelse(logistic_predictions > 0.5, 1, 0), levels = levels(test_data$Outcome))

#Calculating and plot ROC curve for logistic regression
roc_curve_logistic <- roc(test_data$Outcome, logistic_predictions)
## Setting levels: control = 0, case = 1
## Setting direction: controls < cases
auc_score_logistic <- auc(roc_curve_logistic)

#Displaying ROC curve for logistic regression
plot(roc_curve_logistic, main = "ROC Curve (Logistic Regression)", col = "blue", lwd = 2)
abline(a = 0, b = 1, lty = 2, col = "red")  # Diagonal line for reference

#Displaying AUC score for logistic regression
legend("bottomright", legend = paste("AUC =", round(auc_score_logistic, 2)), col = "blue", lty = 1)

#Displaying confusion matrix for logistic regression
logistic_accuracy <- confusionMatrix(logistic_predictions_binary, test_data$Outcome)

#K-Nearest Neighbors (KNN)
knn_model <- train(Outcome ~ ., data = train_data, method = "knn")
knn_predictions <- predict(knn_model, newdata = test_data)

#Converting KNN predictions to binary (0 or 1)
knn_predictions_binary <- factor(knn_predictions, levels = levels(test_data$Outcome))

#Calculating and plot ROC curve for KNN
roc_curve_knn <- roc(test_data$Outcome, as.numeric(knn_predictions))
## Setting levels: control = 0, case = 1
## Setting direction: controls < cases
auc_score_knn <- auc(roc_curve_knn)

#Displaying ROC curve for KNN
plot(roc_curve_knn, main = "ROC Curve (K-Nearest Neighbors)", col = "blue", lwd = 2)
abline(a = 0, b = 1, lty = 2, col = "red")  # Diagonal line for reference

#Displaying AUC score for KNN
legend("bottomright", legend = paste("AUC =", round(auc_score_knn, 2)), col = "blue", lty = 1)
